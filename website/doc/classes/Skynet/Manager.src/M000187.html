    def restart_worker(wpid)
      info "RESTARTING WORKER #{wpid}"
      @mutex.synchronize do
        @workers_running.delete(wpid)
        @workers_restarting += 1
      end
      Process.kill("HUP",wpid)
      sleep Skynet::CONFIG[:WORKER_CHECK_DELAY]
    end
