    def worker_alive?(worker_pid)
      begin                   
        IO.popen("ps -o pid,command -p #{worker_pid}", "r") do |ps|
          return ps.detect {|line| line =~ /worker_type/}
        end
      rescue Errno::ENOENT => e
        return false
      end
      false
    end
