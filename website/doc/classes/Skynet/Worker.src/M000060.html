    def notify_task_complete
      @processed += 1

      mq.write_worker_status(
        @worker_info.merge({
          :task_id       => 0,
          :job_id        => 0,
          :name          => "waiting for #{@worker_type}",
          :processed     => @processed,
          :map_or_reduce => nil,
          :started_at    => Time.now.to_i
        })
      )
    end
