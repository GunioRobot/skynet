    def find_pid_size(file, format=:notpretty)
      begin
        open(file).each { |line|
          if line.index('VmSize')
            temp = line[7..-5].strip.to_f/1000
            return BigDecimal(temp.to_s).truncate(5).to_s('F') if format == :pretty
            return temp
          end
        }
      rescue Exception => e
        warn "ERROR #{e.inspect}"
        '0'
      end
    end
