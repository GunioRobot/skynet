    def self.error_template(message)
      template = {
        :tasktype  => message.tasktype,
        :drburi    => message.drburi, 
        :version   => message.version, 
        :task_id   => message.task_id, 
        :queue_id  => message.queue_id
      }
      fields.collect do |field|
        template[field]
      end
    end
