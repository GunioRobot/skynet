      def stats
        t1 = Time.now
        tasks = list_tasks
        results = list_results
        t2 = Time.now - t1
        p_tasks = tasks.partition {|task| task[9] == 0}
        {:taken_tasks => p_tasks[1].size, :untaken_tasks => p_tasks[0].size, :results => list_results.size, :time => t2.to_f}
      end
