    def reduce_tasks(partitioned_data)
      @reduce_tasks ||= begin
        task_options = {
          :name           => reduce_name,
          :process        => @reduce,
          :map_or_reduce  => :reduce,
          :result_timeout => reduce_timeout,
          :retry          => reduce_retry || Skynet::CONFIG[:DEFAULT_REDUCE_RETRY]
        }
        Skynet::TaskIterator.new(task_options, partitioned_data)
      end
    end
