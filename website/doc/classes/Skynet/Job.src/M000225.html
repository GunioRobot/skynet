                def tasks_to_messages(tasks)
      if tasks.is_a?(Skynet::TaskIterator)
        tasks = tasks.to_a
      elsif not tasks.is_a?(Array)
        tasks = [tasks]
      end

      tasks.collect do |task|
        Skynet::Message.new_task_message(task,self)
      end
          end
