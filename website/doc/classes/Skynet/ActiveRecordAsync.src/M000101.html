  def self.map(datas)
    datas.each do |data| 
      begin            
        model = data[:model_class].constantize.find(data[:model_id])
        if data[:opts]
          model.send(data[:method].to_sym, YAML.load(data[:opts]))
        else
          model.send(data[:method].to_sym)
        end
        model.save if data[:save]
      rescue Exception => e
        error "Error in #{self} #{e.inspect}"
      end
    end
    return
  end
