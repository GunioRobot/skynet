    def self.find(*args)
      if not args.first.is_a?(Hash)
        args.shift
      end       
      if args.nil? or args.empty?
        args = {} 
      else
        args = *args
      end
      new(:find_args => args, :batch_size => args.delete(:batch_size), :model_class => args.delete(:model_class))
    end
