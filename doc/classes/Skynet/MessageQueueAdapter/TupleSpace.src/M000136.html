      def get_worker_version
        begin
          message = Skynet::WorkerVersionMessage.new(read(Skynet::WorkerVersionMessage.template,0.00001))
          if message
            curver = message.version
          else
            curver=0
          end
        rescue Skynet::RequestExpiredError => e
          curver = 0
        end
        curver
      end
