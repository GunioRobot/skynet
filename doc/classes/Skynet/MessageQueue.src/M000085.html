    def self.start_or_connect(adapter_class = self.adapter_class)
      begin
        mq = new
      rescue Skynet::ConnectionError
        if self.adapter == :tuplespace
          pid = fork do
            exec("skynet_tuplespace_server start")
          end
          sleep 5
        end
        new
      end
    end
