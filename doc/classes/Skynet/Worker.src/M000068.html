    def ok_to_mem_check?
      return true if @@ok_to_mem_check == true
      return false if @@ok_to_mem_check == :notok
      if File.exists?('/proc/self/status')
      @@lastmem ||= get_memory_size.to_i
      return @@ok_to_mem_check = true
      else
        @@ok_to_mem_check = :notok
        return false
      end
    end
