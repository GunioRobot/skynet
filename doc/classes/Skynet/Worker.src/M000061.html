    def notify_worker_stop
      info "Worker #{process_id} stopping..."
      wq.write_worker_status(
      @worker_info.merge({
        :task_id       => 0,
        :job_id        => 0,
        :name          => "waiting for #{@worker_type}",
        :processed     => @processed,
        :process_id    => nil,
        :map_or_reduce => nil,
        :started_at    => Time.now.to_i
        })
      )
    end
