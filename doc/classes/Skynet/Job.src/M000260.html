    def map=(map)
      reset!
      return unless map
      if map.class == String or map.class == Class
        @map = map.to_s
      elsif map.is_a?(Proc)
        @map = map        
      else
        raise BadMapOrReduceError.new("#{self.class}.map accepts a class name or a proc. Got #{map}") 
      end      
    end
