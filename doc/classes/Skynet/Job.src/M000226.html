    def map_enqueue      
      task_ids             = []
      map_tasks            = self.map_tasks
      self.use_local_queue = map_local?
      if map_tasks        
        number_of_tasks = 0
        map_tasks.each do |task|
          number_of_tasks += 1
          enqueue_messages(tasks_to_messages(task))
        end            
      end
      return number_of_tasks
    end
